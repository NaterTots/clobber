tree("Root")
	fallback
		tree("RunAwayLittleBoy")
		tree("ChasePlayer")
		tree("Patrol")
		Wait(0.2)

tree("RunAwayLittleBoy")
	while LowHealth
		sequence
			IsTargetNear
			SetSafeDestination
			RunAway

tree("ChasePlayer")
	fallback
		tree("Chase")
		not Wait(0.4)

tree("Chase")
	//Chase the player while they are in range
	while IsTargetNear
		sequence
			ChaseTheTarget
			not LowHealth

tree("Patrol")
	random
		tree("RandomPatrol")
		tree("CirclePatrol")
		//tree("NormalPatrol") - not quite reday for prime time

tree("CirclePatrol")
	while not IsTargetNear
			ExecuteCirclePatrol

tree("RandomPatrol")
	while not IsTargetNear
		sequence
			SetDestination_Random
			MoveToDestination
