tree("Root")
	fallback
		tree("RunAwayLittleBoy")
		tree("ChasePlayer")
		tree("Patrol")
		Wait(0.2)

tree("RunAwayLittleBoy")
	while LowHealth
		sequence
			CanSeeThePlayer
			SetDestination_Safe
			while not NearMoveDestination
				MoveToDestination

tree("ChasePlayer")
	fallback
		tree("Chase")
		not Wait(0.4)

tree("Chase")
	//Chase the player while they are in range
	while CanSeeThePlayer
		sequence
			SetChaseTarget_Player
			ChaseTheTargetUntilCanSee
			Attack
			not LowHealth

tree("Patrol")
	random
		tree("RandomPatrol")
		tree("CirclePatrol")
		//tree("NormalPatrol") - not quite reday for prime time

tree("CirclePatrol")
	while not CanSeeThePlayer
			ExecuteCirclePatrol

tree("RandomPatrol")
	while not CanSeeThePlayer
		sequence
			SetDestination_Random
			MoveToDestination
